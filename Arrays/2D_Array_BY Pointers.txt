#include<stdio.h>
#include<stdlib.h>


int main()
{
    int n;
    scanf("%d",&n);
    int *a;
    a=(int*)malloc(n*n*sizeof(int));
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            scanf("%d",a + i*n +j);
        }
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            printf("%d ",*(a + i*n +j));
        }
        printf("\n");
    }
    return 0;
}
